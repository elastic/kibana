## Description

This package contains a component for rendering and manipulating the form containing the setting fields in the Advanced Settings UI.


## Usage

```tsx
const saveConfig = async (changes: Record<string, UnsavedFieldChange<SettingType>>) => {
  const arr = Object.entries(changes).map(([key, value]) => {
    settingsStart.client.set(key, value.unsavedValue);
  });
  return Promise.all(arr);
};

return <Form fields={fields} save={saveConfig} />;
```
